namespace LC.BLL.Services
{
    public class TransactionService : ITransactionService
    {
        private readonly IRepository<Income> _incomeRepository;
        private readonly IRepository<Expense> _expenseRepository;

        // Мы запрашиваем репозитории через конструктор
        public TransactionService(IRepository<Income> incomeRepository, IRepository<Expense> expenseRepository)
        {
            _incomeRepository = incomeRepository;
            _expenseRepository = expenseRepository;
        }

        public async Task AddIncomeAsync(Income income)
        {
            // Здесь можно добавить проверку, например: if (income.Sum <= 0) ошибка
            await _incomeRepository.AddAsync(income);
        }

        public async Task AddExpenseAsync(Expense expense)
        {
            await _expenseRepository.AddAsync(expense);
        }

        public async Task<IEnumerable<TransactionDto>> GetRecentTransactionsAsync()
        {
            var incomes = await _incomeRepository.GetAllAsync();
            var expenses = await _expenseRepository.GetAllAsync();

            var list = new List<TransactionDto>();

            // Превращаем доходы в DTO
            foreach (var item in incomes)
            {
                list.Add(new TransactionDto
                {
                    Id = item.Id,
                    Date = item.Date ?? DateOnly.FromDateTime(System.DateTime.Now),
                    Operation = "Доход",
                    Category = item.Incomecategory?.Category ?? "Без категории",
                    Amount = item.Sum ?? 0,
                    Color = "#00FF00" // Зеленый
                });
            }

            // Превращаем расходы в DTO
            foreach (var item in expenses)
            {
                list.Add(new TransactionDto
                {
                    Id = item.Id,
                    Date = item.Date ?? DateOnly.FromDateTime(System.DateTime.Now),
                    Operation = "Расход",
                    Category = item.Category?.Name ?? "Без категории",
                    Amount = (item.Sum ?? 0) * -1, // Расход показываем с минусом
                    Color = "#FF0000" // Красный
                });
            }

            // Сортируем по дате (свежие сверху) и берем последние 20
            return list.OrderByDescending(x => x.Date).Take(20);
        }
    }
}

namespace LC.DAL.Repositories
{
    public class EfRepository<T> : IRepository<T> where T : class
    {
        private readonly FamilybudgetdbContext _context;

        public EfRepository(FamilybudgetdbContext context)
        {
            _context = context;
        }

        public async Task<T> GetByIdAsync(int id)
        {
            return await _context.Set<T>().FindAsync(id);
        }

        public async Task<IEnumerable<T>> GetAllAsync()
        {
            return await _context.Set<T>().ToListAsync();
        }

        public async Task AddAsync(T entity)
        {
            await _context.Set<T>().AddAsync(entity);
            await _context.SaveChangesAsync();
        }

        public async Task UpdateAsync(T entity)
        {
            _context.Entry(entity).State = EntityState.Modified;
            await _context.SaveChangesAsync();
        }

        public async Task DeleteAsync(T entity)
        {
            _context.Set<T>().Remove(entity);
            await _context.SaveChangesAsync();
        }
    }
}

namespace LepreCoinsApp.ViewModels
{
    public class BaseViewModel : INotifyPropertyChanged
    {
        public event PropertyChangedEventHandler PropertyChanged;

        // Метод для обновления значения свойства и уведомления интерфейса
        protected void OnPropertyChanged([CallerMemberName] string name = null)
        {
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(name));
        }

        protected bool SetProperty<T>(ref T backingStore, T value, [CallerMemberName] string propertyName = "", System.Action onChanged = null)
        {
            if (System.Collections.Generic.EqualityComparer<T>.Default.Equals(backingStore, value))
                return false;

            backingStore = value;
            onChanged?.Invoke();
            OnPropertyChanged(propertyName);
            return true;
        }
    }
}

namespace LepreCoinsApp.ViewModels
{
    public class ReportsViewModel : BaseViewModel
    {
        private readonly ITransactionService _transactionService;

        // Коллекция, которая автоматически обновляет список на экране
        public ObservableCollection<TransactionDto> Transactions { get; } = new();

        public ReportsViewModel(ITransactionService transactionService)
        {
            _transactionService = transactionService;
            // Загружаем данные при создании
            LoadDataCommand = new Command(async () => await LoadDataAsync());
        }

        public ICommand LoadDataCommand { get; }

        public async Task LoadDataAsync()
        {
            // Получаем данные из сервиса
            var list = await _transactionService.GetRecentTransactionsAsync();

            Transactions.Clear();
            foreach (var item in list)
            {
                Transactions.Add(item);
            }
        }
    }
}

using Microsoft.Maui.Controls;
using System.Threading.Tasks;

namespace LepreCoinsApp.Pages
{
    public partial class MainPage : ContentPage
    {
        public MainPage()
        {
            InitializeComponent();
        }

        protected override async void OnAppearing()
        {
            base.OnAppearing();

            // Сбрасываем плитки в исходное состояние
            var tiles = new[] { IncomeTile, ExpenseTile, SavingsTile, RecentTile };
            foreach (var tile in tiles)
                ResetTile(tile);

            // Анимация появления плиток каскадом
            uint delay = 0;
            foreach (var tile in tiles)
            {
                await AnimateTile(tile, delay);
                delay += 100; // небольшая задержка между плитками
            }
        }

        // Сброс плитки в начальное состояние
        private void ResetTile(View tile)
        {
            tile.Opacity = 0;
            tile.TranslationY = 50;
            tile.Scale = 0.95; // начальный размер чуть меньше
        }

        // Анимация появления плитки
        private async Task AnimateTile(View tile, uint delay)
        {
            await Task.Delay((int)delay);

            await Task.WhenAll(
                tile.FadeToAsync(1, 350, Easing.CubicInOut),
                tile.TranslateToAsync(0, 0, 350, Easing.CubicOut),
                tile.ScaleToAsync(1.0, 350, Easing.CubicOut)
            );
        }

        // Эффект при наведении курсора (увеличение плитки)
        private async void OnTileEntered(object sender, PointerEventArgs e)
        {
            if (sender is View tile)
            {
                await tile.ScaleToAsync(1.03, 120, Easing.CubicOut);
                tile.Opacity = 0.97; // лёгкий эффект фокуса
            }
        }

        // Эффект при уходе курсора (возврат к нормальному размеру)
        private async void OnTileExited(object sender, PointerEventArgs e)
        {
            if (sender is View tile)
            {
                await tile.ScaleToAsync(1.0, 120, Easing.CubicOut);
                tile.Opacity = 1.0;
            }
        }
    }
}
